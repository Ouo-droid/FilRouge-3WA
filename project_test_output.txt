\n
use Kentec\App\Model\State;\n
use Kentec\App\Repository\AbsenceRepository;\n
use Kentec\App\Repository\ProjectRepository;\n
use Kentec\App\Repository\TaskRepository;\n
use Kentec\Kernel\Database\Repository;\n
use Kentec\Kernel\Http\AbstractController;\n
use Kentec\Kernel\Security\InputValidator;\n
use Kentec\Kernel\Security\Security;\n
use OpenApi\Attributes as OA;\n
\n
class ProjectController extends AbstractController\n
{\n
\n
    // Affichage des projets en mode "vue"\n
    #[OA\Get(\n
        path: '/projects',\n
        summary: 'Show projects list page',\n
        tags: ['Projects'],\n
        responses: [\n
            new OA\Response(response: 200, description: 'Projects page'),\n
        ]\n
    )]\n
    final public function index(): void\n
    {\n
        $projectRepo = new ProjectRepository();\n
        $userRepo    = new Repository(\Kentec\App\Model\User::class);\n
        $taskRepo    = new TaskRepository();\n
        $stateRepo   = new Repository(State::class);\n
\n
        $currentUser = Security::getUser();\n
        $userRole    = $currentUser ? ($currentUser->getRoleName() ?? 'USER') : 'USER';\n
\n
        // USER : seulement les projets sur lesquels il a au moins une tâche assignée\n
        if ($userRole === 'USER' && $currentUser) {\n
            $projects = $projectRepo->findActiveByUserId($currentUser->getId());\n
        } else {\n
            $projects = $projectRepo->findAllActive();\n
        }\n
\n
        // Récupération de tous les utilisateurs une seule fois\n
        $users = $userRepo->getAll();\n
\n
        $usersById = [];\n
        foreach ($users as $user) {\n
            $usersById[$user->getId()] = $user;\n
        }\n
\n
        // Récupération des tâches et états\n
        $tasks = $taskRepo->getAll();\n
\n
        $states = $stateRepo->getAll();\n
\n
        // Construire un tableau des états par ID\n
        $statesById = [];\n
        if ($states) {\n
            foreach ($states as $state) {\n
                $statesById[$state->getId()] = $state;\n
            }\n
        }\n
\n
        // Récupérer les assignations task->user via la table de liaison\n
        $assignments    = $taskRepo->findAllUserAssignments();\n
        $taskDevelopers = [];\n
        foreach ($assignments as $assignment) {\n
            $taskDevelopers[$assignment['task_id']] = $assignment['user_id'];\n
        }\n
\n
        // Ajout des infos calculées aux projets\n
        foreach ($projects as &$project) {\n
            // Normalisation des clés (DB lowercase vs Code camelCase)\n
            $project['beginDate'] = $project['beginDate'] ?? $project['begindate'] ?? null;\n
            $project['theoricalDeadLine'] = $project['theoricalDeadLine'] ?? $project['theoricaldeadline'] ?? null;\n
            $project['realDeadLine'] = $project['realDeadLine'] ?? $project['realdeadline'] ?? null;\n
\n
            // Hydratation du chef de projet\n
            if (!empty($project['project_manager_id']) && isset($usersById[$project['project_manager_id']])) {\n
                $manager = $usersById[$project['project_manager_id']];\n
                $project['manager_firstname'] = $manager->getFirstname();\n
                $project['manager_lastname'] = $manager->getLastname();\n
                $project['manager_object'] = $manager;\n
            }\n
\n
            // Calculs basés sur les tâches\n
            $projectTasks = [];\n
            $completedTasksCount = 0;\n
            $participants = [];\n
\n
            // Ajouter le manager comme participant de base\n
            if (!empty($project['project_manager_id']) && isset($usersById[$project['project_manager_id']])) {\n
                $user = $usersById[$project['project_manager_id']];\n
                $participants[$project['project_manager_id']] = $user->getFirstname() . ' ' . $user->getLastname();\n
            }\n
\n
            if ($tasks) {\n
                foreach ($tasks as $task) {\n
                    if ($task->getProjectId() == $project['id']) {\n
                        $projectTasks[] = $task;\n
\n
                        // Vérifier le statut\n
                        $taskStateId = $task->getStateId();\n
                        if ($taskStateId && isset($statesById[$taskStateId])) {\n
                            $stateName = $statesById[$taskStateId]->getName();\n
                            if (false !== stripos($stateName, 'termin')\n
                                || false !== stripos($stateName, 'done')\n
                                || false !== stripos($stateName, 'clos')\n
                                || false !== stripos($stateName, 'fini')) {\n
                                ++$completedTasksCount;\n
                            }\n
                        }\n
\n
                        // Ajouter le développeur assigné comme participant (via usertaskREL)\n
                        $devId = $taskDevelopers[$task->getId()] ?? null;\n
                        if ($devId && isset($usersById[$devId])) {\n
                            $user = $usersById[$devId];\n
                            $participants[$devId] = $user->getFirstname() . ' ' . $user->getLastname();\n
                        }\n
                    }\n
                }\n
            }\n
\n
            $totalTasks = count($projectTasks);\n
            $project['task_count'] = $totalTasks;\n
            $project['progress'] = $totalTasks > 0 ? round(($completedTasksCount / $totalTasks) * 100) : 0;\n
            $project['participants_count'] = count($participants);\n
            $project['participants_list'] = array_values($participants);\n
\n
            // Nom du statut du projet\n
            $projectStateId = $project['state_id'] ?? null;\n
            if ($projectStateId && isset($statesById[$projectStateId])) {\n
                $project['state_name'] = $statesById[$projectStateId]->getName();\n
            } else {\n
                $project['state_name'] = 'En attente';\n
            }\n
        }\n
        unset($project);\n
\n
        // Comptage des projets par état (pour les onglets de filtre)\n
        $stateCounts = [];\n
        foreach ($statesById as $sid => $state) {\n
            $stateCounts[$sid] = 0;\n
        }\n
        $upcomingDeadlines = 0;\n
        $now      = new \DateTime();\n
        $nextWeek = (new \DateTime())->modify('+7 days');\n
        foreach ($projects as $project) {\n
            $sid = $project['state_id'] ?? null;\n
            if ($sid && isset($stateCounts[$sid])) {\n
                ++$stateCounts[$sid];\n
            }\n
            $deadline = $project['theoricalDeadLine'] ?? null;\n
            if ($deadline) {\n
                try {\n
                    $dl = new \DateTime($deadline);\n
                    if ($dl > $now && $dl <= $nextWeek) {\n
                        ++$upcomingDeadlines;\n
                    }\n
                } catch (\Exception $e) {\n
                    // date invalide, on ignore\n
                }\n
            }\n
        }\n
\n
        // Comptage des projets par statut\n
        $statusStats = [\n
            'en_attente' => 0,\n
            'en_cours'   => 0,\n
            'termine'    => 0,\n
            'retarde'    => 0,\n
            'annule'     => 0,\n
        ];\n
\n
        foreach ($projects as $project) {\n
            $stateId = $project['state_id'] ?? null;\n
            if ($stateId && isset($statesById[$stateId])) {\n
                $stateName = strtolower($statesById[$stateId]->getName());\n
                if (str_contains($stateName, 'annul')) {\n
                    ++$statusStats['annule'];\n
                } elseif (str_contains($stateName, 'retard')) {\n
                    ++$statusStats['retarde'];\n
                } elseif (str_contains($stateName, 'termin') || str_contains($stateName, 'done') || str_contains($stateName, 'fini')) {\n
                    ++$statusStats['termine'];\n
                } elseif (str_contains($stateName, 'cours') || str_contains($stateName, 'progress')) {\n
                    ++$statusStats['en_cours'];\n
                } else {\n
                    ++$statusStats['en_attente'];\n
                }\n
            } else {\n
                ++$statusStats['en_attente'];\n
            }\n
        }\n
\n
        $this->render('project/project.php', [\n
            'pageTitle'         => 'Projets',\n
            'projects'          => $projects,\n
            'userRole'          => $userRole,\n
            'statusStats'       => $statusStats,\n
            'states'            => $statesById,\n
            'stateCounts'       => $stateCounts,\n
            'upcomingDeadlines' => $upcomingDeadlines,\n
        ]);\n
    }\n
\n
    // Affichage des détails d'un projet\n
    #[OA\Get(\n
        path: '/project/{id}',\n
        summary: 'Show project details page',\n
        tags: ['Projects'],\n
        parameters: [\n
            new OA\Parameter(name: 'id', in: 'path', required: true, schema: new OA\Schema(type: 'string', format: 'uuid')),\n
        ],\n
        responses: [\n
            new OA\Response(response: 200, description: 'Project details page'),\n
            new OA\Response(response: 302, description: 'Redirect to projects list if not found'),\n
        ]\n
    )]\n
    final public function show(string $id): void\n
    {\n
        if (!InputValidator::validateUuid($id)) {\n
            header('Location: /projects');\n
            exit;\n
        }\n
\n
        $currentUser = Security::getUser();\n
        $userRole    = $currentUser ? ($currentUser->getRoleName() ?? 'USER') : 'USER';\n
        $canCreate   = in_array($userRole, ['ADMIN', 'CDP', 'PDG'], true);\n
        $canDelete   = in_array($userRole, ['ADMIN', 'PDG'], true);\n
\n
        $projectRepo = new ProjectRepository();\n
        $userRepo    = new Repository(\Kentec\App\Model\User::class);\n
        $taskRepo    = new TaskRepository();\n
        $stateRepo   = new Repository(State::class);\n
\n
        // Récupérer le projet actif\n
        $project = $projectRepo->findActiveById($id);\n
\n
        if ($project === null) {\n
            header('Location: /projects');\n
            exit;\n
        }\n
\n
        // Contrôle d'accès : USER ne peut voir que les projets sur lesquels il a une tâche\n
        if ($userRole === 'USER' && $currentUser) {\n
            if (!$taskRepo->isUserAssignedToProject($id, $currentUser->getId())) {\n
                header('Location: /projects');\n
                exit;\n
            }\n
        }\n
\n
        // Normalisation des clés\n
        $project['beginDate'] = $project['beginDate'] ?? $project['begindate'] ?? null;\n
        $project['theoricalDeadLine'] = $project['theoricalDeadLine'] ?? $project['theoricaldeadline'] ?? null;\n
        $project['realDeadLine'] = $project['realDeadLine'] ?? $project['realdeadline'] ?? null;\n
\n
        // Récupérer les utilisateurs\n
        $users = $userRepo->getAll();\n
        $usersById = [];\n
        foreach ($users as $user) {\n
            $usersById[$user->getId()] = $user;\n
        }\n
\n
        // Récupérer les états\n
        $states = $stateRepo->getAll();\n
        $statesById = [];\n
        $completedStateId = null;\n
        if ($states) {\n
            foreach ($states as $state) {\n
                $statesById[$state->getId()] = $state;\n
                if (false !== stripos($state->getName(), 'termin')\n
                    || false !== stripos($state->getName(), 'done')\n
                    || false !== stripos($state->getName(), 'clos')\n
                    || false !== stripos($state->getName(), 'fini')) {\n
                    $completedStateId = $state->getId();\n
                }\n
            }\n
        }\n
\n
        // Hydratation manager\n
        if (!empty($project['project_manager_id']) && isset($usersById[$project['project_manager_id']])) {\n
            $project['manager_object'] = $usersById[$project['project_manager_id']];\n
        }\n
\n
        // Récupérer les tâches du projet\n
        $tasks = $taskRepo->getByAttributes(['project_id' => $id]);\n
        if (!is_array($tasks)) {\n
            $tasks = $tasks ? [$tasks] : [];\n
        }\n
\n
        // Récupérer les assignations task->user via la table de liaison\n
        $assignments    = $taskRepo->findAllUserAssignments();\n
        $taskDevelopers = [];\n
        foreach ($assignments as $assignment) {\n
            $taskDevelopers[$assignment['task_id']] = $assignment['user_id'];\n
        }\n
\n
        // Calculs Statistiques\n
        $stats = [\n
            'total' => count($tasks),\n
            'completed' => 0,\n
            'in_progress' => 0,\n
            'todo' => 0,\n
            'progress' => 0,\n
        ];\n
\n
        $absenceRepo   = new AbsenceRepository();\n
        $absenceRows   = $absenceRepo->findActiveTodayWithDates();\n
        $absentUserIds = array_column($absenceRows, 'user_id');\n
        $absenceByUser = [];\n
        foreach ($absenceRows as $row) {\n
            $absenceByUser[$row['user_id']] = $row;\n
        }\n
\n
        $team = [];\n
        // Ajouter le manager à l'équipe\n
        if (!empty($project['project_manager_id']) && isset($usersById[$project['project_manager_id']])) {\n
            $userId = $project['project_manager_id'];\n
            $team[$userId] = [\n
                'user'         => $usersById[$userId],\n
                'role'         => 'Chef de projet',\n
                'active_tasks' => 0,\n
                'is_absent'    => in_array($userId, $absentUserIds, true),\n
                'absence'      => $absenceByUser[$userId] ?? null,\n
            ];\n
        }\n
\n
        foreach ($tasks as $task) {\n
            $stateId = $task->getStateId();\n
\n
            // Stats par état\n
            $stateName = isset($statesById[$stateId]) ? $statesById[$stateId]->getName() : '';\n
\n
            if ($stateId == $completedStateId) {\n
                ++$stats['completed'];\n
            } elseif (false !== stripos($stateName, 'todo')\n
                      || false !== stripos($stateName, 'faire')\n
                      || false !== stripos($stateName, 'attente')\n
                      || false !== stripos($stateName, 'backlog')) {\n
                ++$stats['todo'];\n
            } else {\n
                ++$stats['in_progress'];\n
            }\n
\n
            // Membres de l'équipe via usertaskREL\n
            $devId = $taskDevelopers[$task->getId()] ?? null;\n
            if ($devId && isset($usersById[$devId])) {\n
                if (!isset($team[$devId])) {\n
                    $team[$devId] = [\n
                        'user'         => $usersById[$devId],\n
                        'role'         => 'Développeur',\n
                        'active_tasks' => 0,\n
                        'is_absent'    => in_array($devId, $absentUserIds, true),\n
                        'absence'      => $absenceByUser[$devId] ?? null,\n
                    ];\n
                }\n
                $isCompleted = (false !== stripos($stateName, 'termin')\n
                                || false !== stripos($stateName, 'done')\n
                                || false !== stripos($stateName, 'clos')\n
                                || false !== stripos($stateName, 'fini'));\n
                if (!$isCompleted) {\n
                    ++$team[$devId]['active_tasks'];\n
                }\n
            }\n
        }\n
\n
        $stats['progress'] = $stats['total'] > 0 ? round(($stats['completed'] / $stats['total']) * 100) : 0;\n
\n
        // Effort totals\n
        $effortRequired = 0.0;\n
        $effortMade     = 0.0;\n
        foreach ($tasks as $task) {\n
            $effortRequired += (float) ($task->getEffortrequired() ?? 0);\n
            $effortMade     += (float) ($task->getEffortmade()     ?? 0);\n
        }\n
        $stats['effort_required'] = round($effortRequired, 2);\n
        $stats['effort_made']     = round($effortMade, 2);\n
\n
        $this->render('project/details.php', [\n
            'project'        => $project,\n
            'tasks'          => $tasks,\n
            'stats'          => $stats,\n
            'team'           => $team,\n
            'states'         => $statesById,\n
            'taskDevelopers' => $taskDevelopers,\n
            'userRole'       => $userRole,\n
            'canCreate'      => $canCreate,\n
            'canDelete'      => $canDelete,\n
            'allUsers'       => array_values($usersById),\n
        ]);\n
    }\n
\n

    #[OA\Get(\n

        tags: ['Projects'],\n
        responses: [\n
            new OA\Response(\n
                response: 200,\n
                description: 'Successful response',\n

                    properties: [\n
                        new OA\Property(property: 'projects', type: 'array', items: new OA\Items(ref: '#/components/schemas/Project')),\n
                    ]\n
                )\n
            ),\n
        ]\n
    )]\n
    final public function getApiProjects(): void\n
    {\n
        if ('GET' !== $_SERVER['REQUEST_METHOD']) {\n
            $this->jsonError('La méthode HTTP doit être GET', 405);\n
            return;\n
        }\n
\n
        try {\n
            $projectRepo = new ProjectRepository();\n
            $userRepo    = new Repository(\Kentec\App\Model\User::class);\n
\n
            $currentUser = Security::getUser();\n
            $userRole    = $currentUser ? ($currentUser->getRoleName() ?? 'USER') : 'USER';\n
\n
            // USER : seulement les projets où il a au moins une tâche assignée\n
            if ($userRole === 'USER' && $currentUser) {\n
                $projects = $projectRepo->findActiveByUserId($currentUser->getId());\n
            } else {\n
                $projects = $projectRepo->findAllActive();\n
            }\n
\n
            $users = $userRepo->getAll();\n
            $usersById = [];\n
            foreach ($users as $user) {\n
                $usersById[$user->getId()] = $user;\n
            }\n
\n
            foreach ($projects as &$project) {\n
                // Normalisation des clés\n
                $project['beginDate'] = $project['beginDate'] ?? $project['begindate'] ?? null;\n
                $project['theoricalDeadLine'] = $project['theoricalDeadLine'] ?? $project['theoricaldeadline'] ?? null;\n
                $project['realDeadLine'] = $project['realDeadLine'] ?? $project['realdeadline'] ?? null;\n
\n
                if (!empty($project['project_manager_id']) && isset($usersById[$project['project_manager_id']])) {\n
                    $manager = $usersById[$project['project_manager_id']];\n
                    $project['manager_firstname'] = $manager->getFirstname();\n
                    $project['manager_lastname'] = $manager->getLastname();\n
                    $project['manager_email'] = $manager->getEmail();\n
                }\n
            }\n
\n
            $this->jsonSuccess(['projects' => $projects ?? []]);\n
        } catch (\Exception $e) {\n
            $this->jsonError($e->getMessage(), 500);\n
        }\n
    }\n
\n
    #[OA\Get(\n
        path: '/api/project/{projectId}',\n
        summary: 'Get project by ID (JSON)',\n
        tags: ['Projects'],\n
        parameters: [\n
            new OA\Parameter(name: 'projectId', in: 'path', required: true, schema: new OA\Schema(type: 'string', format: 'uuid')),\n
        ],\n
        responses: [\n
            new OA\Response(\n
                response: 200,\n
                description: 'Successful response',\n
                content: new OA\JsonContent(\n
                    properties: [\n
                        new OA\Property(property: 'project', ref: '#/components/schemas/Project'),\n
                    ]\n
                )\n
            ),\n
            new OA\Response(response: 404, description: 'Project not found'),\n
        ]\n
    )]\n
    final public function getApiProject(string $projectId): void\n
    {\n
        if ('GET' !== $_SERVER['REQUEST_METHOD']) {\n
            $this->jsonError('La méthode HTTP doit être GET', 405);\n
            return;\n
        }\n
\n
        try {\n
            if (!InputValidator::validateUuid($projectId)) {\n
                $this->jsonError('Invalid UUID format', 400);\n
                return;\n
            }\n
\n
            $taskRepo    = new TaskRepository();\n
            $projectRepo = new ProjectRepository();\n
\n
            // USER : vérifier qu'il a au moins une tâche sur ce projet\n
            $currentUser = Security::getUser();\n
            $userRole    = $currentUser ? ($currentUser->getRoleName() ?? 'USER') : 'USER';\n
\n
            if ($userRole === 'USER' && $currentUser) {\n
                if (!$taskRepo->isUserAssignedToProject($projectId, $currentUser->getId())) {\n
                    $this->jsonError('Accès non autorisé', 403);\n
                    return;\n
                }\n
            }\n
\n
            $row = $projectRepo->findWithDetailsById($projectId);\n
\n
            if ($row === null) {\n
                $this->jsonError('Projet introuvable', 404);\n
                return;\n
            }\n
\n
            $project = Project::fromDatabaseArray($row);\n
\n
            // Calcul d'effort via méthode dédiée\n
            $effort = $this->calculerEffortProjet($taskRepo, $projectId);\n
\n
            $this->jsonSuccess([\n
                'project'            => $project,\n
                'totalEffortRequired' => $effort['totalEffortRequired'],\n
                'totalEffortMade'     => $effort['totalEffortMade'],\n
                'totalTasks'          => $effort['totalTasks'],\n
            ]);\n
        } catch (\Exception $e) {\n
            $this->jsonError($e->getMessage(), 500);\n
        }\n
    }\n
\n
    /**\n
     * Calcule les totaux d'effort pour un projet.\n
     * Utilise COALESCE pour retourner 0 si aucune tâche n'est associée.\n
     */\n
    private function calculerEffortProjet(TaskRepository $taskRepo, string $projectId): array\n
    {\n
        return $taskRepo->findEffortByProjectId($projectId);\n
    }\n
\n
    final public function dynamicalProjects(): void\n
    {\n
        $projectRepo = new ProjectRepository();\n
        $projects    = $projectRepo->findAllActive();\n
        $this->render('project/dynamicalProjects.php', ['projects' => $projects]);\n
    }\n

    #[OA\Delete(\n
        path: '/api/delete/project/{projectId}',\n
        summary: 'Soft-delete project (isactive = false)',\n
        tags: ['Projects'],\n
        parameters: [\n
            new OA\Parameter(name: 'projectId', in: 'path', required: true, schema: new OA\Schema(type: 'string', format: 'uuid')),\n
        ],\n
        responses: [\n
            new OA\Response(response: 200, description: 'Project soft-deleted'),\n
            new OA\Response(response: 404, description: 'Project not found'),\n
        ]\n
    )]\n
    final public function deleteApiProject(string $projectId): void\n
    {\n
        $token = $_SERVER['HTTP_X_CSRF_TOKEN'] ?? '';\n
        if (!Security::verifyCsrfToken($token)) {\n
            $this->jsonError('Requête invalide (token CSRF manquant ou expiré).', 403);\n
            return;\n
        }\n
\n
        if ('DELETE' !== $_SERVER['REQUEST_METHOD']) {\n
            $this->jsonError('La méthode HTTP doit être DELETE', 405);\n
            return;\n
        }\n
\n
        try {\n
            if (!InputValidator::validateUuid($projectId)) {\n
                $this->jsonError('Invalid Project ID (must be a valid UUID)', 400);\n
                return;\n
            }\n
\n
            $currentUser = Security::getUser();\n
\n
            $projectRepo = new ProjectRepository();\n
            $project     = $projectRepo->getById($projectId);\n
\n
            if (!$project) {\n
                $this->jsonError('Project not found', 404);\n
                return;\n
            }\n
\n
            // Soft delete : isactive = false (jamais de DELETE physique)\n
            $projectRepo->softDelete($projectId, $currentUser?->getId());\n
\n
            // Archiver également toutes les tâches associées au projet\n
            $taskRepo = new TaskRepository();\n
            $taskRepo->archiveByProject($projectId, $currentUser?->getId());\n
\n
            $this->jsonSuccess(['message' => 'Projet et ses tâches archivés avec succès']);\n
        } catch (\Exception $e) {\n
            $this->jsonError($e->getMessage(), 500);\n
        }\n
    }\n

    #[OA\Post(\n
        path: '/api/add/project',\n
        summary: 'Add new project (JSON)',\n
        tags: ['Projects'],\n
        requestBody: new OA\RequestBody(\n
            required: true,\n
            content: new OA\JsonContent(\n
                required: ['name', 'beginDate', 'theoreticalDeadline'],\n
                properties: [\n
                    new OA\Property(property: 'name', type: 'string'),\n
                    new OA\Property(property: 'description', type: 'string'),\n
                    new OA\Property(property: 'clientId', type: 'string', format: 'uuid'),\n
                    new OA\Property(property: 'projectManagerId', type: 'string', format: 'uuid'),\n
                    new OA\Property(property: 'beginDate', type: 'string', format: 'date'),\n
                    new OA\Property(property: 'theoreticalDeadline', type: 'string', format: 'date'),\n
                    new OA\Property(property: 'realDeadline', type: 'string', format: 'date'),\n
                ]\n
            )\n
        ),\n
        responses: [\n
            new OA\Response(response: 200, description: 'Project created'),\n
            new OA\Response(response: 400, description: 'Invalid input'),\n
        ]\n
    )]\n
    final public function addApiProject(): void\n
    {\n

    #[OA\Get(\n
        path: '/projects',\n
        summary: 'Show projects list page',\n
        tags: ['Projects'],\n
        responses: [\n
            new OA\Response(response: 200, description: 'Projects page'),\n
        ]\n
    )]\n
    final public function index(): void\n
    {\n
        $projectRepo = new ProjectRepository();\n
        $userRepo    = new Repository(\Kentec\App\Model\User::class);\n
        $taskRepo    = new TaskRepository();\n
        $stateRepo   = new Repository(State::class);\n
\n
        $currentUser = Security::getUser();\n
        $userRole    = $currentUser ? ($currentUser->getRoleName() ?? 'USER') : 'USER';\n
\n
        // USER : seulement les projets sur lesquels il a au moins une tâche assignée\n
        if ($userRole === 'USER' && $currentUser) {\n
            $projects = $projectRepo->findActiveByUserId($currentUser->getId());\n
        } else {\n
            $projects = $projectRepo->findAllActive();\n
        }\n
\n
        // Récupération de tous les utilisateurs une seule fois\n
        $users = $userRepo->getAll();\n
\n
        $usersById = [];\n
        foreach ($users as $user) {\n
            $usersById[$user->getId()] = $user;\n
        }\n
\n
        // Récupération des tâches et états\n
        $tasks = $taskRepo->getAll();\n
\n
        $states = $stateRepo->getAll();\n
\n
        // Construire un tableau des états par ID\n
        $statesById = [];\n
        if ($states) {\n
            foreach ($states as $state) {\n
                $statesById[$state->getId()] = $state;\n
            }\n
        }\n
\n
        // Récupérer les assignations task->user via la table de liaison\n
        $assignments    = $taskRepo->findAllUserAssignments();\n
        $taskDevelopers = [];\n
        foreach ($assignments as $assignment) {\n
            $taskDevelopers[$assignment['task_id']] = $assignment['user_id'];\n
        }\n
\n
        // Ajout des infos calculées aux projets\n
        foreach ($projects as &$project) {\n
            // Normalisation des clés (DB lowercase vs Code camelCase)\n
            $project['beginDate'] = $project['beginDate'] ?? $project['begindate'] ?? null;\n
            $project['theoricalDeadLine'] = $project['theoricalDeadLine'] ?? $project['theoricaldeadline'] ?? null;\n
            $project['realDeadLine'] = $project['realDeadLine'] ?? $project['realdeadline'] ?? null;\n
\n
            // Hydratation du chef de projet\n
            if (!empty($project['project_manager_id']) && isset($usersById[$project['project_manager_id']])) {\n
                $manager = $usersById[$project['project_manager_id']];\n
                $project['manager_firstname'] = $manager->getFirstname();\n
                $project['manager_lastname'] = $manager->getLastname();\n
                $project['manager_object'] = $manager;\n
            }\n
\n
            // Calculs basés sur les tâches\n
            $projectTasks = [];\n
            $completedTasksCount = 0;\n
            $participants = [];\n
\n
            // Ajouter le manager comme participant de base\n
            if (!empty($project['project_manager_id']) && isset($usersById[$project['project_manager_id']])) {\n
                $user = $usersById[$project['project_manager_id']];\n
                $participants[$project['project_manager_id']] = $user->getFirstname() . ' ' . $user->getLastname();\n
            }\n
\n
            if ($tasks) {\n
                foreach ($tasks as $task) {\n
                    if ($task->getProjectId() == $project['id']) {\n
                        $projectTasks[] = $task;\n
\n
                        // Vérifier le statut\n
                        $taskStateId = $task->getStateId();\n
                        if ($taskStateId && isset($statesById[$taskStateId])) {\n
                            $stateName = $statesById[$taskStateId]->getName();\n
                            if (false !== stripos($stateName, 'termin')\n
                                || false !== stripos($stateName, 'done')\n
                                || false !== stripos($stateName, 'clos')\n
                                || false !== stripos($stateName, 'fini')) {\n
                                ++$completedTasksCount;\n
                            }\n
                        }\n
\n
                        // Ajouter le développeur assigné comme participant (via usertaskREL)\n
                        $devId = $taskDevelopers[$task->getId()] ?? null;\n
                        if ($devId && isset($usersById[$devId])) {\n
                            $user = $usersById[$devId];\n
                            $participants[$devId] = $user->getFirstname() . ' ' . $user->getLastname();\n
                        }\n
                    }\n
                }\n
            }\n
\n
            $totalTasks = count($projectTasks);\n
            $project['task_count'] = $totalTasks;\n
            $project['progress'] = $totalTasks > 0 ? round(($completedTasksCount / $totalTasks) * 100) : 0;\n
            $project['participants_count'] = count($participants);\n
            $project['participants_list'] = array_values($participants);\n
\n
            // Nom du statut du projet\n
            $projectStateId = $project['state_id'] ?? null;\n
            if ($projectStateId && isset($statesById[$projectStateId])) {\n
                $project['state_name'] = $statesById[$projectStateId]->getName();\n
            } else {\n
                $project['state_name'] = 'En attente';\n
            }\n
        }\n
        unset($project);\n
\n
        // Comptage des projets par état (pour les onglets de filtre)\n
        $stateCounts = [];\n
        foreach ($statesById as $sid => $state) {\n
            $stateCounts[$sid] = 0;\n
        }\n
        $upcomingDeadlines = 0;\n
        $now      = new \DateTime();\n
        $nextWeek = (new \DateTime())->modify('+7 days');\n
        foreach ($projects as $project) {\n
            $sid = $project['state_id'] ?? null;\n
            if ($sid && isset($stateCounts[$sid])) {\n
                ++$stateCounts[$sid];\n
            }\n
            $deadline = $project['theoricalDeadLine'] ?? null;\n
            if ($deadline) {\n
                try {\n
                    $dl = new \DateTime($deadline);\n
                    if ($dl > $now && $dl <= $nextWeek) {\n
                        ++$upcomingDeadlines;\n
                    }\n
                } catch (\Exception $e) {\n
                    // date invalide, on ignore\n
                }\n
            }\n
        }\n
\n
        // Comptage des projets par statut\n
        $statusStats = [\n
            'en_attente' => 0,\n
            'en_cours'   => 0,\n
            'termine'    => 0,\n
            'retarde'    => 0,\n
            'annule'     => 0,\n
        ];\n
\n
        foreach ($projects as $project) {\n
            $stateId = $project['state_id'] ?? null;\n
            if ($stateId && isset($statesById[$stateId])) {\n
                $stateName = strtolower($statesById[$stateId]->getName());\n
                if (str_contains($stateName, 'annul')) {\n
                    ++$statusStats['annule'];\n
                } elseif (str_contains($stateName, 'retard')) {\n
                    ++$statusStats['retarde'];\n
                } elseif (str_contains($stateName, 'termin') || str_contains($stateName, 'done') || str_contains($stateName, 'fini')) {\n
                    ++$statusStats['termine'];\n
                } elseif (str_contains($stateName, 'cours') || str_contains($stateName, 'progress')) {\n
                    ++$statusStats['en_cours'];\n
                } else {\n
                    ++$statusStats['en_attente'];\n
                }\n
            } else {\n
                ++$statusStats['en_attente'];\n
            }\n
        }\n
\n
        $this->render('project/project.php', [\n
            'pageTitle'         => 'Projets',\n
            'projects'          => $projects,\n
            'userRole'          => $userRole,\n
            'statusStats'       => $statusStats,\n
            'states'            => $statesById,\n
            'stateCounts'       => $stateCounts,\n
            'upcomingDeadlines' => $upcomingDeadlines,\n
        ]);\n
    }\n

    // Affichage des détails d'un projet\n
    #[OA\Get(\n
        path: '/project/{id}',\n
        summary: 'Show project details page',\n
        tags: ['Projects'],\n
        parameters: [\n
            new OA\Parameter(name: 'id', in: 'path', required: true, schema: new OA\Schema(type: 'string', format: 'uuid')),\n
        ],\n
        responses: [\n
            new OA\Response(response: 200, description: 'Project details page'),\n
            new OA\Response(response: 302, description: 'Redirect to projects list if not found'),\n
        ]\n
    )]\n
    final public function show(string $id): void\n
    {\n
        if (!InputValidator::validateUuid($id)) {\n
            header('Location: /projects');\n
            exit;\n
        }\n
\n
        $currentUser = Security::getUser();\n
        $userRole    = $currentUser ? ($currentUser->getRoleName() ?? 'USER') : 'USER';\n
        $canCreate   = in_array($userRole, ['ADMIN', 'CDP', 'PDG'], true);\n
        $canDelete   = in_array($userRole, ['ADMIN', 'PDG'], true);\n
\n
        $projectRepo = new ProjectRepository();\n
        $userRepo    = new Repository(\Kentec\App\Model\User::class);\n
        $taskRepo    = new TaskRepository();\n
        $stateRepo   = new Repository(State::class);\n
\n
        // Récupérer le projet actif\n
        $project = $projectRepo->findActiveById($id);\n
\n
        if ($project === null) {\n
            header('Location: /projects');\n
            exit;\n
        }\n
\n
        // Contrôle d'accès : USER ne peut voir que les projets sur lesquels il a une tâche\n
        if ($userRole === 'USER' && $currentUser) {\n
            if (!$taskRepo->isUserAssignedToProject($id, $currentUser->getId())) {\n
                header('Location: /projects');\n
                exit;\n
            }\n
        }\n
\n
        // Normalisation des clés\n
        $project['beginDate'] = $project['beginDate'] ?? $project['begindate'] ?? null;\n
        $project['theoricalDeadLine'] = $project['theoricalDeadLine'] ?? $project['theoricaldeadline'] ?? null;\n
        $project['realDeadLine'] = $project['realDeadLine'] ?? $project['realdeadline'] ?? null;\n
\n
        // Récupérer les utilisateurs\n
        $users = $userRepo->getAll();\n
        $usersById = [];\n
        foreach ($users as $user) {\n
            $usersById[$user->getId()] = $user;\n
        }\n
\n
        // Récupérer les états\n
        $states = $stateRepo->getAll();\n
        $statesById = [];\n
        $completedStateId = null;\n
        if ($states) {\n
            foreach ($states as $state) {\n
                $statesById[$state->getId()] = $state;\n
                if (false !== stripos($state->getName(), 'termin')\n
                    || false !== stripos($state->getName(), 'done')\n
                    || false !== stripos($state->getName(), 'clos')\n
                    || false !== stripos($state->getName(), 'fini')) {\n
                    $completedStateId = $state->getId();\n
                }\n
            }\n
        }\n
\n
        // Hydratation manager\n
        if (!empty($project['project_manager_id']) && isset($usersById[$project['project_manager_id']])) {\n
            $project['manager_object'] = $usersById[$project['project_manager_id']];\n
        }\n
\n
        // Récupérer les tâches du projet\n
        $tasks = $taskRepo->getByAttributes(['project_id' => $id]);\n
        if (!is_array($tasks)) {\n
            $tasks = $tasks ? [$tasks] : [];\n
        }\n
\n
        // Récupérer les assignations task->user via la table de liaison\n
        $assignments    = $taskRepo->findAllUserAssignments();\n
        $taskDevelopers = [];\n
        foreach ($assignments as $assignment) {\n
            $taskDevelopers[$assignment['task_id']] = $assignment['user_id'];\n
        }\n
\n
        // Calculs Statistiques\n
        $stats = [\n
            'total' => count($tasks),\n
            'completed' => 0,\n
            'in_progress' => 0,\n
            'todo' => 0,\n
            'progress' => 0,\n
        ];\n
\n
        $absenceRepo   = new AbsenceRepository();\n
        $absenceRows   = $absenceRepo->findActiveTodayWithDates();\n
        $absentUserIds = array_column($absenceRows, 'user_id');\n
        $absenceByUser = [];\n
        foreach ($absenceRows as $row) {\n
            $absenceByUser[$row['user_id']] = $row;\n
        }\n
\n
        $team = [];\n
        // Ajouter le manager à l'équipe\n
        if (!empty($project['project_manager_id']) && isset($usersById[$project['project_manager_id']])) {\n
            $userId = $project['project_manager_id'];\n
            $team[$userId] = [\n
                'user'         => $usersById[$userId],\n
                'role'         => 'Chef de projet',\n
                'active_tasks' => 0,\n
                'is_absent'    => in_array($userId, $absentUserIds, true),\n
                'absence'      => $absenceByUser[$userId] ?? null,\n
            ];\n
        }\n
\n
        foreach ($tasks as $task) {\n
            $stateId = $task->getStateId();\n
\n
            // Stats par état\n
            $stateName = isset($statesById[$stateId]) ? $statesById[$stateId]->getName() : '';\n
\n
            if ($stateId == $completedStateId) {\n
                ++$stats['completed'];\n
            } elseif (false !== stripos($stateName, 'todo')\n
                      || false !== stripos($stateName, 'faire')\n
                      || false !== stripos($stateName, 'attente')\n
                      || false !== stripos($stateName, 'backlog')) {\n
                ++$stats['todo'];\n
            } else {\n
                ++$stats['in_progress'];\n
            }\n
\n
            // Membres de l'équipe via usertaskREL\n
            $devId = $taskDevelopers[$task->getId()] ?? null;\n
            if ($devId && isset($usersById[$devId])) {\n
                if (!isset($team[$devId])) {\n
                    $team[$devId] = [\n
                        'user'         => $usersById[$devId],\n
                        'role'         => 'Développeur',\n
                        'active_tasks' => 0,\n
                        'is_absent'    => in_array($devId, $absentUserIds, true),\n
                        'absence'      => $absenceByUser[$devId] ?? null,\n
                    ];\n
                }\n
                $isCompleted = (false !== stripos($stateName, 'termin')\n
                                || false !== stripos($stateName, 'done')\n
                                || false !== stripos($stateName, 'clos')\n
                                || false !== stripos($stateName, 'fini'));\n
                if (!$isCompleted) {\n
                    ++$team[$devId]['active_tasks'];\n
                }\n
            }\n
        }\n
\n
        $stats['progress'] = $stats['total'] > 0 ? round(($stats['completed'] / $stats['total']) * 100) : 0;\n
\n
        // Effort totals\n
        $effortRequired = 0.0;\n
        $effortMade     = 0.0;\n
        foreach ($tasks as $task) {\n
            $effortRequired += (float) ($task->getEffortrequired() ?? 0);\n
            $effortMade     += (float) ($task->getEffortmade()     ?? 0);\n
        }\n
        $stats['effort_required'] = round($effortRequired, 2);\n
        $stats['effort_made']     = round($effortMade, 2);\n
\n
        $this->render('project/details.php', [\n
            'project'        => $project,\n
            'tasks'          => $tasks,\n
            'stats'          => $stats,\n
            'team'           => $team,\n
            'states'         => $statesById,\n
            'taskDevelopers' => $taskDevelopers,\n
            'userRole'       => $userRole,\n
            'canCreate'      => $canCreate,\n
            'canDelete'      => $canDelete,\n
            'allUsers'       => array_values($usersById),\n
        ]);\n
    }\n

    #[OA\Get(\n
        path: '/api/project/{projectId}',\n
        summary: 'Get project by ID (JSON)',\n
        tags: ['Projects'],\n
        parameters: [\n
            new OA\Parameter(name: 'projectId', in: 'path', required: true, schema: new OA\Schema(type: 'string', format: 'uuid')),\n
        ],\n
        responses: [\n
            new OA\Response(\n
                response: 200,\n
                description: 'Successful response',\n
                content: new OA\JsonContent(\n
                    properties: [\n
                        new OA\Property(property: 'project', ref: '#/components/schemas/Project'),\n
                    ]\n
                )\n
            ),\n
            new OA\Response(response: 404, description: 'Project not found'),\n
        ]\n
    )]\n
    final public function getApiProject(string $projectId): void\n
    {\n
        if ('GET' !== $_SERVER['REQUEST_METHOD']) {\n
            $this->jsonError('La méthode HTTP doit être GET', 405);\n
            return;\n
        }\n
\n
        try {\n
            if (!InputValidator::validateUuid($projectId)) {\n
                $this->jsonError('Invalid UUID format', 400);\n
                return;\n
            }\n
\n
            $taskRepo    = new TaskRepository();\n
            $projectRepo = new ProjectRepository();\n
\n
            // USER : vérifier qu'il a au moins une tâche sur ce projet\n
            $currentUser = Security::getUser();\n
            $userRole    = $currentUser ? ($currentUser->getRoleName() ?? 'USER') : 'USER';\n
\n
            if ($userRole === 'USER' && $currentUser) {\n
                if (!$taskRepo->isUserAssignedToProject($projectId, $currentUser->getId())) {\n
                    $this->jsonError('Accès non autorisé', 403);\n
                    return;\n
                }\n
            }\n
\n
            $row = $projectRepo->findWithDetailsById($projectId);\n
\n
            if ($row === null) {\n
                $this->jsonError('Projet introuvable', 404);\n
                return;\n
            }\n
\n
            $project = Project::fromDatabaseArray($row);\n
\n
            // Calcul d'effort via méthode dédiée\n
            $effort = $this->calculerEffortProjet($taskRepo, $projectId);\n
\n
            $this->jsonSuccess([\n
                'project'            => $project,\n
                'totalEffortRequired' => $effort['totalEffortRequired'],\n
                'totalEffortMade'     => $effort['totalEffortMade'],\n
                'totalTasks'          => $effort['totalTasks'],\n
            ]);\n
        } catch (\Exception $e) {\n
            $this->jsonError($e->getMessage(), 500);\n
        }\n
    }\n

    /**\n
     * Calcule les totaux d'effort pour un projet.\n
     * Utilise COALESCE pour retourner 0 si aucune tâche n'est associée.\n
     */\n
    private function calculerEffortProjet(TaskRepository $taskRepo, string $projectId): array\n
    {\n
        return $taskRepo->findEffortByProjectId($projectId);\n
    }\n

    #[OA\Delete(\n
        path: '/api/delete/project/{projectId}',\n
        summary: 'Soft-delete project (isactive = false)',\n
        tags: ['Projects'],\n
        parameters: [\n
            new OA\Parameter(name: 'projectId', in: 'path', required: true, schema: new OA\Schema(type: 'string', format: 'uuid')),\n
        ],\n
        responses: [\n
            new OA\Response(response: 200, description: 'Project soft-deleted'),\n
            new OA\Response(response: 404, description: 'Project not found'),\n
        ]\n
    )]\n
    final public function deleteApiProject(string $projectId): void\n
    {\n
        $token = $_SERVER['HTTP_X_CSRF_TOKEN'] ?? '';\n
        if (!Security::verifyCsrfToken($token)) {\n
            $this->jsonError('Requête invalide (token CSRF manquant ou expiré).', 403);\n
            return;\n
        }\n
\n
        if ('DELETE' !== $_SERVER['REQUEST_METHOD']) {\n
            $this->jsonError('La méthode HTTP doit être DELETE', 405);\n
            return;\n
        }\n
\n
        try {\n
            if (!InputValidator::validateUuid($projectId)) {\n
                $this->jsonError('Invalid Project ID (must be a valid UUID)', 400);\n
                return;\n
            }\n
\n
            $currentUser = Security::getUser();\n
\n
            $projectRepo = new ProjectRepository();\n
            $project     = $projectRepo->getById($projectId);\n
\n
            if (!$project) {\n
                $this->jsonError('Project not found', 404);\n
                return;\n
            }\n
\n
            // Soft delete : isactive = false (jamais de DELETE physique)\n
            $projectRepo->softDelete($projectId, $currentUser?->getId());\n
\n
            // Archiver également toutes les tâches associées au projet\n
            $taskRepo = new TaskRepository();\n
            $taskRepo->archiveByProject($projectId, $currentUser?->getId());\n
\n
            $this->jsonSuccess(['message' => 'Projet et ses tâches archivés avec succès']);\n
        } catch (\Exception $e) {\n
            $this->jsonError($e->getMessage(), 500);\n
        }\n
    }\n

        summary: 'Add new project (JSON)',\n
        tags: ['Projects'],\n
        requestBody: new OA\RequestBody(\n

    final public function addApiProject(): void\n
    {\n
        $token = $_SERVER['HTTP_X_CSRF_TOKEN'] ?? '';\n
        if (!Security::verifyCsrfToken($token)) {\n
            $this->jsonError('Requête invalide (token CSRF manquant ou expiré).', 403);\n
            return;\n
        }\n
\n
        if ('POST' !== $_SERVER['REQUEST_METHOD']) {\n
            $this->jsonError('La méthode HTTP doit être POST', 405);\n
            return;\n
        }\n
\n
        try {\n
            $inputData = json_decode(file_get_contents('php://input'), true);\n
